\subsubsection{Vector addition}

$$
\CYAN{\vec{v}} + \ORNG{\vec{w}} = \begin{bmatrix}
    \CYAN{v_1} \\ \CYAN{v_2}
\end{bmatrix} + \begin{bmatrix}
    \ORNG{w_1} \\ \ORNG{w_2}
\end{bmatrix} = \begin{bmatrix}
    \CYAN{v_1} + \ORNG{w_1} \\ \CYAN{v_2} + \ORNG{w_2}
\end{bmatrix}
$$
